102        //MM:dd:yyyy

112        final SimpleDateFormat MMddyyyy = new SimpleDateFormat("MM/dd/yyyy");
113        Map<String, Object> map = new HashMap<String, Object>();
114        map.put("date", Calendar.getInstance().getTime());
115        sc.setDateFormat(MMddyyyy);

182    @Test
183    public void testDeserializationCollection() throws Exception {
184        List<Rectangle> rectList = new ArrayList<Rectangle>();
185        for (int i = 0; i < 10; i++) {
186            Rectangle filteredRect = new Rectangle(i, i, "rect" + i);
187            rectList.add(filteredRect);
188        }
189        String serColl = sc.ser(rectList);
190        Type collectionType = new TypeToken<Collection<Rectangle>>() {}.getType();
191        List<Rectangle> resultList0 = (List<Rectangle>) ((org.apache.fulcrum.json.gson.GSONBuilderService)sc).deSerCollection(serColl, collectionType,Rectangle.class);
192        for (int i = 0; i < 10; i++) {
193            assertEquals("deser reread size failed", (i * i), resultList0
194                    .get(i).getSize());
195        }
196    }
