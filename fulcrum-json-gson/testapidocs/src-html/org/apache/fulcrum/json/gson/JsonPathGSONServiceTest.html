062    @Test
063    public void testSerialize() throws Exception
064    {
065        String serJson = sc.ser( new TestClass("mytest") );
066        String cf = ((JsonPrimitive) JsonPath.parse(serJson).read("$.container.cf")).getAsString();// .using(conf)
067        assertEquals("Serialization failed ", "Config.xml", cf);   
068    }

070    @Test
071    public void testDefaultGsonSerializeDate() throws Exception
072    {
073        // default calls with default DateTypeAdapter
074        Map<String,Object> map = new HashMap<String,Object>();
075        Calendar refDate = Calendar.getInstance();
076        map.put( "date", refDate.getTime() );
077
078        //sc.setDateFormat( MMddyyyy );
079        String serJson = sc.ser( map );  
080        System.out.println("serJson:"+ serJson);
081        assertTrue("Serialize with Adapater failed ", serJson.matches( "\\{\"date\":\"\\d\\d:\\d\\d:\\d{4}\"\\}" ));
082
083        Date date = JsonPath.parse(serJson).read("$.date", Date.class);// .using(conf)
084        Calendar parsedDate = Calendar.getInstance();
085        parsedDate.setTime(date);
086        assertEquals("Serialization failed ", refDate.get(Calendar.DATE), parsedDate.get(Calendar.DATE));  
087    }
088    
089    @Test
090    public void testGsonSerializeDate() throws Exception
091    {
092        final SimpleDateFormat MMddyyyy = new SimpleDateFormat( "MM/dd/yyyy");
093        Map<String,Object> map = new HashMap<String,Object>();
094        Calendar refDate = Calendar.getInstance();
095        map.put( "date", refDate.getTime() );
096        //((GSONBuilderService)sc).setDateFormat( "MM/dd/yyyy" );
097        sc.setDateFormat( MMddyyyy );
098        String serJson = sc.ser( map );  
099        System.out.println("serJson:"+ serJson);
100        assertTrue("Serialize with Adapter failed ", serJson.matches( "\\{\"date\":\"\\d\\d/\\d\\d/\\d{4}\"\\}" ));
101        Date date = JsonPath.parse(serJson).read("$.date", Date.class);// .using(conf)
102        Calendar parsedDate = Calendar.getInstance();
103        parsedDate.setTime(date);
104        assertEquals("Serialization failed ", refDate.get(Calendar.DATE), parsedDate.get(Calendar.DATE));  
105    }

107    @Test
108    public void testCollection() throws Exception
109    {
110      List<Rectangle> rectList = new ArrayList<Rectangle>();
111      for ( int i = 0; i < 10; i++ )
112      {
113          Rectangle filteredRect = new Rectangle(i,i,"rect"+i);
114          rectList.add( filteredRect ); 
115      }
116      String result = sc.ser( rectList );
117      assertEquals("collect ser","[{'w':0,'h':0,'name':'rect0'},{'w':1,'h':1,'name':'rect1'},{'w':2,'h':2,'name':'rect2'},{'w':3,'h':3,'name':'rect3'},{'w':4,'h':4,'name':'rect4'},{'w':5,'h':5,'name':'rect5'},{'w':6,'h':6,'name':'rect6'},{'w':7,'h':7,'name':'rect7'},{'w':8,'h':8,'name':'rect8'},{'w':9,'h':9,'name':'rect9'}]",
118              result.replace( '"', '\'' )       );
119      List<Map<String, Object>> beanList2 = JsonPath.parse(result).read("$[-2:]", List.class);
120      assertEquals("Expect 2 Elements failed ", 2, beanList2.size());     
121     }
122    @Test
123    public void testSerializeTypeAdapterForCollection() throws Exception
124    {
125      sc.addAdapter( "Collection Adapter",  ArrayList.class , TypeAdapterForCollection.class );
126      List<Rectangle> rectList = new ArrayList<Rectangle>();
127      for ( int i = 0; i < 10; i++ )
128      {
129          Rectangle filteredRect = new Rectangle(i,i,"rect"+i);
130          rectList.add( filteredRect ); 
131      }
132      String serColl = sc.ser( rectList  );
133      assertEquals("collect ser","[{'rect0':0,'rect1':1,'rect2':4,'rect3':9,'rect4':16,'rect5':25,'rect6':36,'rect7':49,'rect8':64,'rect9':81}]",
134              serColl.replace( '"', '\'' )       );
135      
136      System.out.println("serColl: "+ serColl);
137      TypeRef<List<Rectangle>> typeRef = new TypeRef<List<Rectangle>>() { };
138      List<Rectangle> result = JsonPath.parse(serColl).read("$",typeRef);
139      System.out.println("result: "+ result);
140      int idx = 0;
141      for (Rectangle rect : result) {
142          assertEquals("deser reread size failed", (idx * idx), rect.getSize());
143          idx++;
144      }
145     }
