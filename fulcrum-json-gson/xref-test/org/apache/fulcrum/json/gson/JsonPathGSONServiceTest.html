70      @Test
71      public void testDefaultGsonSerializeDate() throws Exception
72      {
73          // default calls with default DateTypeAdapter
74          Map<String,Object> map = new HashMap<String,Object>();
75          Calendar refDate = Calendar.getInstance();
76          map.put( "date", refDate.getTime() );
77  
78          //sc.setDateFormat( MMddyyyy );
79          String serJson = sc.ser( map );  
80          System.out.println("serJson:"+ serJson);
81          assertTrue("Serialize with Adapater failed ", serJson.matches( "\\{\"date\":\"\\d\\d:\\d\\d:\\d{4}\"\\}" ));
82  
83          Date date = JsonPath.parse(serJson).read("$.date", Date.class);// .using(conf)
84          Calendar parsedDate = Calendar.getInstance();
85          parsedDate.setTime(date);
86          assertEquals("Serialization failed ", refDate.get(Calendar.DATE), parsedDate.get(Calendar.DATE));  
87      }

89      @Test
90      public void testGsonSerializeDate() throws Exception
91      {
92          final SimpleDateFormat MMddyyyy = new SimpleDateFormat( "MM/dd/yyyy");
93          Map<String,Object> map = new HashMap<String,Object>();
94          Calendar refDate = Calendar.getInstance();
95          map.put( "date", refDate.getTime() );
96          //((GSONBuilderService)sc).setDateFormat( "MM/dd/yyyy" );
97          sc.setDateFormat( MMddyyyy );
98          String serJson = sc.ser( map );  
99          System.out.println("serJson:"+ serJson);
100         assertTrue("Serialize with Adapter failed ", serJson.matches( "\\{\"date\":\"\\d\\d/\\d\\d/\\d{4}\"\\}" ));
101         Date date = JsonPath.parse(serJson).read("$.date", Date.class);// .using(conf)
102         Calendar parsedDate = Calendar.getInstance();
103         parsedDate.setTime(date);
104         assertEquals("Serialization failed ", refDate.get(Calendar.DATE), parsedDate.get(Calendar.DATE));  
105     }
