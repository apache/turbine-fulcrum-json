051public class JsonOrgJacksonMapperTest extends BaseUnit5Test {
052    
053    private JsonService sc = null;
054    Logger logger;
055
056    @BeforeEach
057    public void setUp() throws Exception {
058        sc = (JsonService) this.lookup(JsonService.ROLE);
059        sc.addAdapter(null, null,new JsonOrgModule());
060        //((Jackson2MapperService)sc).registerModule(new JsonOrgModule());
061        logger = new Log4JLogger(LogManager.getLogger(getClass().getName()) );
062    }
063    
064    // support for org.json mapping 
065    @Test
066    public void testDeSerToORGJSONCollectionObject() throws Exception {
067        // test array
068         List<Bean> beanResults = new ArrayList<Bean> ( );
069         Bean tu = new Bean();
070         tu.setName("jim jar");
071         beanResults.add(tu);
072         Bean tu2 = new Bean();
073         tu2.setName("jim2 jar2");
074         tu2.setAge(45);
075         beanResults.add(tu2);
076         String[] filterAttr = {"name", "age" };   
077         String filteredSerList = sc.serializeOnlyFilter(beanResults, Bean.class, filterAttr);
078         logger.debug("serList: "+ filteredSerList);
079         JSONArray jsonOrgResult = sc.deSer(filteredSerList, JSONArray.class);//readValue(serList, JSONArray.class);
080         logger.debug("jsonOrgResult: "+ jsonOrgResult.toString(2));
081         assertEquals("jim jar", ((JSONObject)(jsonOrgResult.get(0))).get("name"), "DeSer failed " );
082         assertEquals( 45, ((JSONObject)(jsonOrgResult.get(1))).get("age"), "DeSer failed " );
083    }
084    
085    // support for org.json mapping 
086    @Test
087    public void testSerToORGJSONCollectionObject() throws Exception {
088        // test array
089         List<Bean> beanList = new ArrayList<Bean> ( );
090         Bean tu = new Bean();
091         tu.setName("jim jar");
092         beanList.add(tu);
093         Bean tu2 = new Bean();
094         tu2.setName("jim2 jar2");
095         tu2.setAge(45);
096         beanList.add(tu2);
097         String[] filterAttr = {"name", "age" };
098         JSONArray jsonArray =((Jackson2MapperService)sc).getMapper().convertValue( beanList, JSONArray .class );
099         logger.debug("jsonArray: "+ jsonArray.toString(2));
100         assertEquals("jim jar", ((JSONObject)jsonArray.get( 0 )).get( "name" ), "Get JSONObject from jsonArray failed ");
101    }
102    
103    // support for org.json mapping 
104    @Test
105    public void testSerToORGJSONObject() throws Exception {
106        Bean bean = new Bean();
107        bean.setName("joe");
108        bean.setAge(12);
109        JSONObject jsonObject =((Jackson2MapperService)sc).getMapper().convertValue( bean, JSONObject.class );
110        logger.debug("jsonObject: "+ jsonObject.toString(2));
111        assertEquals("joe", jsonObject.get( "name" ),"Get name from jsonObject failed ");
112    }
113    
114    @Test
115    public void testFilteredBeanAfterDeSerToORGJSONObject() throws Exception {
116        Bean bean = new Bean();
117        bean.setName("joe2");
118        bean.setAge(12);
119        //same as ((Jackson2MapperService)sc).getMapper().convertValue( bean, JSONObject.class );
120        JSONObject jsonObject =((Jackson2MapperService)sc).deSer( bean, JSONObject.class );
121        logger.debug("jsonObject: "+ jsonObject.toString(2));
122        //  without no filter is applied, filter on original bean requires new mapper, bug ?
123        ((Jackson2MapperService)sc).initialize();// need to get fresh mapper
124        String[] filterAttr = {"name" };
125        String filteredBean = sc.serializeOnlyFilter(bean, Bean.class, filterAttr);
126        logger.debug("filteredBean: "+ filteredBean);
127        assertEquals("{\"name\":\"joe2\"}", filteredBean, "Ser filtered Bean failed ");
128    }
129    
130    @Test
131    public void testFilteredJSONObjectAfterDeSerToORGJSONObject() throws Exception {
132        Bean bean = new Bean();
133        bean.setName("joe3");
134        bean.setAge(22);
135        // alternatively two step instead of calling convertValue, which does not affect filtering
136        String serBean = sc.ser( bean, Bean.class );
137        JSONObject jsonObject = sc.deSer( serBean, JSONObject.class );
138        logger.debug("jsonObject: "+ jsonObject.toString(2));
139        // retrieve hashmap to allow filtering by SimpleBeanPropertyFilter 
140        String serJsonObject = sc.ser( jsonObject, JSONObject.class );
141        Map mapOfJsonObject = sc.deSer( serJsonObject, HashMap.class );
142        String[] filterAttr = {"name" };
143        String filteredJsonObject = sc.serializeOnlyFilter(mapOfJsonObject, filterAttr);
144        logger.debug("filteredJsonObject: "+ filteredJsonObject);
145        assertEquals("{\"name\":\"joe3\"}", filteredJsonObject, "Ser filtered JSONObject failed ");
146        // works without init
147        String filteredBean = sc.serializeOnlyFilter(bean, Bean.class, filterAttr);
148        logger.debug("filteredBean: "+ filteredBean);
149        assertEquals("{\"name\":\"joe3\"}", filteredBean, "Ser filtered Bean failed ");
150    }
