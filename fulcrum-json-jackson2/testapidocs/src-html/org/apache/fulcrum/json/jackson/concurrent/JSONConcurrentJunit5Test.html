106        @ParameterizedTest
107        @MethodSource("multipleRandStream")
108        public void one(int randStreamInt) throws InterruptedException {
109            String result = doJob(randStreamInt, "name");
110            assertTrue(!result.contains("org.apache.fulcrum.json.jackson.example.Bean"), "Result does contain type");
111            assertTrue(!result.contains("java.util.ArrayList"), "Result does contain type");
112            fOne = Thread.currentThread();
113        }

115        @ParameterizedTest
116        @MethodSource("multipleRandStream")
117        public void two(int randStreamInt) throws InterruptedException {
118            String result = doJob(randStreamInt, "name", "age");
119            assertTrue(!result.contains("org.apache.fulcrum.json.jackson.example.Bean"), "Result does contain type");
120            assertTrue(!result.contains("java.util.ArrayList"), "Result does contain type");
121            fTwo = Thread.currentThread();
122        }

124        @ParameterizedTest
125        @MethodSource("multipleRandStream")
126        public void three(int randStreamInt) throws InterruptedException, JsonProcessingException {
127            ObjectMapper mapper = customMapper(true);
128            // ((Jackson2MapperService) jsonService).setMapper(mapper);
129            // String result = doTaskJob("name", "age","profession");
130            String result = doFilteredJob(mapper, randStreamInt, new String[] { "age", "profession" });
131            assertTrue(result.contains("org.apache.fulcrum.json.jackson.example.Bean"),
132                    "Result does not contain type, but it shouldn't");
133            assertTrue(result.contains("java.util.ArrayList"), "Result does not contain type, but it shouldn't");
134            assertTrue(!result.contains("\"name\""), "Result should not contain attribute name");
135            fThree = Thread.currentThread();
136        }

138        private String doJob(int randStreamInt, String... filtereIds) {
139            List<Bean> tasks = getBeans(randStreamInt);
140            String result = getJson(tasks, Bean.class, true, filtereIds);
141            System.out.println("ser result (id=" + Thread.currentThread() + "):" + result);
142
143            List<String> fidsArr = Arrays.asList(filtereIds);
144            fidsArr.forEach(fidId -> {
145                assertTrue(result.contains("\"" + fidId + "\""), "Result does not contain (" + fidId + ")");
146                assertTrue(result.contains(randStreamInt + ""), result + " does not contain " + randStreamInt + ")");
147            });
148            return result;
149        }
150
151        private String doFilteredJob(ObjectMapper mapper, int randStreamInt, String... filtereIds)
152                throws JsonProcessingException {
153            List<Bean> tasks = getBeans(randStreamInt);
154            String result = serFiltered(tasks, Bean.class, mapper, filtereIds);
155            System.out.println("ser result (id=" + Thread.currentThread() + "):" + result);
156            List<String> fidsArr = Arrays.asList(filtereIds);
157            fidsArr.forEach(fidId -> {
158                assertTrue(result.contains("\"" + fidId + "\""), "Result does not contain (" + fidId + ")");
159                assertTrue(result.contains(randStreamInt + ""), result + " does not contain " + randStreamInt + ")");
160            });
161            return result;
162        }
163
164        private String serFiltered(List<?> list, Class<?> filterClass, ObjectMapper mapper, String[] filterAttr)
165                throws JsonProcessingException {
166            Collection<AnnotationIntrospector> ais = ((AnnotationIntrospectorPair) mapper.getSerializationConfig()
167                    .getAnnotationIntrospector()).allIntrospectors();
168            // activate filtering
169            ais.stream().filter(ai -> ai instanceof SimpleNameIntrospector)
170                    .forEach(ai -> ((SimpleNameIntrospector) ai).setFilteredClasses(filterClass));
171            PropertyFilter pf = SimpleBeanPropertyFilter.filterOutAllExcept(filterAttr);
172            SimpleFilterProvider filter = new SimpleFilterProvider();
173            filter.setDefaultFilter(pf);
174            return mapper.writer(filter).writeValueAsString(list);
175        }
176
177        private ObjectMapper customMapper(boolean withType) {
178            ObjectMapper objectMapper = null;
179            if (withType) {
180                // without copy tests will fail in testsRunInParallel
181                objectMapper = ((Jackson2MapperService) jsonService).getMapper().copy();
182                objectMapper.enableDefaultTypingAsProperty(DefaultTyping.NON_FINAL, "type");
183                AnnotationIntrospector ai = objectMapper.getSerializationConfig().getAnnotationIntrospector();
184                AnnotationIntrospector siai = new SimpleNameIntrospector();
185                AnnotationIntrospector pair = new AnnotationIntrospectorPair(siai, ai);
186                objectMapper.setAnnotationIntrospector(pair);
187            } else {
188                objectMapper = ((Jackson2MapperService) jsonService).getMapper();
189            }
190            return objectMapper;
191        }
192
193        // randNum is in any property -> unconditionally checkable
194        private List<Bean> getBeans(int randNum) {
195            List<Bean> beans = new ArrayList<Bean>();
196            IntStream.of(50).forEach(i -> {
197                Bean dct = new Bean();
198                dct.setName("title" + i + "_" + randNum);
199                dct.setAge(randNum);
200                dct.profession = "prof" + i + "_" + randNum;
201                beans.add(dct);
202            });
203            return beans;
204        }
205
206        private String getJson(List<?> list, Class clazz, Boolean refresh, String... props) {
207            try {
208                return jsonService.serializeOnlyFilter(list, clazz, refresh, props);
209            } catch (Exception e) {
210                System.out.println(e.getMessage());
211                fail();
212            }
213            return null;
214        }

230    @RepeatedTest(10)
231    public void testsRunInParallel(TestReporter testReporter) {
232        // Parallel among methods in a class in one thread pool
233        final LauncherDiscoveryRequest request = LauncherDiscoveryRequestBuilder.request()
234                .selectors(selectClass(JSONBeansTests.class))
235                // does not matter:
236                // if JSONBeansTests is set to per_class
237                // (junit.jupiter.testinstance.lifecycle.default) test methods may reuse threads
238                // or
239                // if it is default per_method.
240                // .configurationParameter("junit.jupiter.testinstance.lifecycle.default",
241                // "per_class")
242                .configurationParameter("junit.jupiter.execution.parallel.enabled", "true").build();
243        TestExecutionSummary summary = defaultExecute(request);
244        defaultAsserts(testReporter, summary);
245
246        // this method may be executed multiple times, any method may get a reused thread at the end, no assert
247//              assertThat(fOne, is(not(fTwo))); 
248//              assertThat(fTwo, is(not(fThree)));
249//              assertThat(fOne, is(not(fThree)));
250    }

253     * 
254     * @param testReporter to reportfrom assets

256    @RepeatedTest(3)
257    public void testsRunInSameThread(TestReporter testReporter) {
258        // Parallel among methods in a class in one thread pool
259        final LauncherDiscoveryRequest request = LauncherDiscoveryRequestBuilder.request()
260                .selectors(selectClass(JSONBeansTests.class))
261                // .configurationParameter("junit.jupiter.testinstance.lifecycle.default",
262                // "per_class")
263                .build();
264        TestExecutionSummary summary = defaultExecute(request);
265        defaultAsserts(testReporter, summary);
266        assertThat(fOne, is(fTwo));
267        assertThat(fTwo, is(fThree));
268        assertThat(fOne, is(fThree));
269    }
270
271    private TestExecutionSummary defaultExecute(final LauncherDiscoveryRequest request) {
272        final Launcher launcher = LauncherFactory.create();
273        final SummaryGeneratingListener listener = new SummaryGeneratingListener();
274
275        launcher.registerTestExecutionListeners(listener);
276        launcher.execute(request);
277
278        return listener.getSummary();
279    }

281    /**
282     * 
283     * @param testReporter to report summary, threads and failures
284     * @param summary to get failures and test count
285     */
286    private void defaultAsserts(TestReporter testReporter, TestExecutionSummary summary) {
287        List<Failure> failures = summary.getFailures();
288        assertNotNull(summary.getTestsFoundCount() > 0, "No Tests found");
289
290        testReporter.publishEntry("summary getTestsStartedCount:" + summary.getTestsStartedCount());
291        testReporter.publishEntry("summary getTestsSucceededCount:" + summary.getTestsSucceededCount());
292        testReporter.publishEntry("fOne:" + fOne);
293        testReporter.publishEntry("fTwo:" + fTwo);
294        testReporter.publishEntry("fThree:" + fThree);
295
296        failures.forEach(failure -> {
297            testReporter.publishEntry("failure - " + failure.getException());
298            testReporter.publishEntry("failure at - " + failure.getTestIdentifier());
299        });
300        long testFoundCount = summary.getTestsFoundCount();
301
302                assertNotNull(fOne);
303                assertNotNull(fTwo);
304                assertNotNull(fThree);
305
306        assertEquals(testFoundCount, summary.getTestsSucceededCount(),
307                "Not exactly " + testFoundCount + " Tests succeessfull");
308        assertEquals(0, summary.getTestsFailedCount(), "No failed tests");
309        assertTrue(failures.isEmpty(), "Parallel Test failures not empty");
310    }
