146    private <T> T checkDeserialization(String serJson, Class<T> target, Class mixin) throws Exception {
147        sc.addAdapter("TestClass Adapter", target, mixin);
148        T result = sc.deSer(serJson,target);
149        //logger.debug("result:"+ result + " is of type "+ target.getName());
150        assertTrue(target.isAssignableFrom(result.getClass()), "result instance check");
151        return result;
152     }
153    
154    private <U> Collection<U> checkDeserCollection(String serJson,Class<? extends Collection> collClass, Class entryClass) throws Exception {
155          Collection<U> result = ((Jackson2MapperService) sc).deSerCollectionWithType(serJson, collClass, entryClass);
156          assertTrue( 
157                  collClass.isAssignableFrom(result.getClass()),
158                  "result instance check failed for result class "+ result.getClass() + " and target class: "+ collClass);
159          return result;
160    }

163    public void testDeserializationCollectionWithFilter() throws Exception {
164
165        List<Bean> beanList = new ArrayList<Bean>();
166        for (int i = 0; i < 10; i++) {
167            Bean filteredBean = new Bean();
168            filteredBean.setName("joe" + i);
169            filteredBean.setAge(i);
170            beanList.add(filteredBean);
171        }
172        String result = sc.serializeOnlyFilter(beanList, Bean.class, "name",
173                "age");
174        //System.out.println("res:" +result);
175        // serialized json has type for list (ArrayList) and type for elements (Bean) -> TypeReference could be used
176        TypeReference typeRef = new TypeReference<List<Bean>>(){};
177        List<Bean> beanList2 = (List<Bean>) ((Jackson2MapperService) sc)
178                .deSerCollection(result, typeRef, Bean.class);
179        assertTrue(beanList2.size() == 10);
180        for (Bean bean : beanList2) {
181            assertEquals(Bean.class, bean.getClass());
182        }
183    }
