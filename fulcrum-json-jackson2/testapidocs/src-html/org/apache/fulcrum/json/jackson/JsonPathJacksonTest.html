064    @BeforeEach
065    public void setUp() throws Exception {
066        sc = (JsonService) this.lookup(JsonService.ROLE);
067        logger = new Log4JLogger(LogManager.getLogger(getClass().getName()) );
068        try {
069            Configuration conf = Configuration.defaultConfiguration();
070            logger.debug("jayway jsonpath conf:"+ conf.jsonProvider());
071            assertEquals( JacksonJsonProvider.class.getName(), conf.jsonProvider().getClass().getName(),
072                          "Jackson JsonPath JsonProvider match failed ");
073            
074            logger.debug("Jackson2MapperService.mapper:"+ ((Jackson2MapperService)sc).getMapper()  + " confjsonProvider:" + conf.jsonProvider());
075            
076            assertTrue(conf.jsonProvider() instanceof JacksonJsonProvider, 
077                       "JsonProvider is not a JacksonJsonProvider ");
078            
079            assertEquals(((Jackson2MapperService)sc).getMapper(), ((JacksonJsonProvider)conf.jsonProvider()).getObjectMapper(), 
080                         "JacksonJsonProvider mapper is not configured mapper");
081        } catch (Throwable e) {
082            if (e.getCause() != null && e.getCause() instanceof ClassNotFoundException) { 
083                logger.error(e.getMessage(), e.getCause());
084                fail("Check correct initialization with useJsonPath = true):");
085            } else {
086                fail(e.getMessage());
087            }
088        }
089
090    }

100    @Test
101    public void testSerializeDateWithDefaultDateFormat() throws Exception {
102        Map<String, Date> map = new HashMap<String, Date>();
103        Calendar refDate = Calendar.getInstance();
104        map.put("date", refDate.getTime());
105        String serJson = sc.ser(map);
106        //System.out.println("serJson date: "+ serJson);
107        Date date = JsonPath.parse(serJson).read("$.date", Date.class);// .using(conf)
108        Calendar parsedDate = Calendar.getInstance();
109        parsedDate.setTime(date);
110        assertEquals(refDate.get(Calendar.DATE), parsedDate.get(Calendar.DATE), "Serialization failed ");
111    }

113    @Test
114    public void testDeserializationCollectionWithFilter() throws Exception {
115
116        List<Bean> beanList = new ArrayList<Bean>();
117        for (int i = 0; i < 10; i++) {
118            Bean filteredBean = new Bean();
119            filteredBean.setName("joe" + i);
120            filteredBean.setAge(i);
121            beanList.add(filteredBean);
122        }
123        String result = sc.serializeOnlyFilter(beanList, Bean.class, "name",   "age");
124        //System.out.println("bean list: "+ result);
125        
126        Bean joe2 = JsonPath.parse(result).read("$[2]", Bean.class);
127        assertEquals("joe2", joe2.getName(), 
128                     "DeSer failed ");
129        // could not map to typed list
130        List<Map<String, Object>> beanList2 = JsonPath.parse(result).read("$[-2:]", List.class);
131        assertEquals(2, beanList2.size(), 
132                     "Expect 2 Elements failed ");
133        //System.out.println("bean list result: "+ beanList2);
134        assertEquals("joe9", beanList2.get(1).get("name"), 
135                     "DeSer failed ");
136    }

151    @Test
152    public void testDeserializationCollection() throws Exception {
153        List<Rectangle> rectList = new ArrayList<Rectangle>(); 
154        for (int i = 0; i < 10; i++) {
155            Rectangle filteredRect = new Rectangle(i, i, "rect" + i);
156            rectList.add(filteredRect);
157        }
158        String serColl = sc.ser(rectList);
159        logger.debug("serColl: "+ serColl);
160        TypeRef<List<Rectangle>> typeRef = new TypeRef<List<Rectangle>>() { };
161        List<Rectangle> result = JsonPath.parse(serColl).read("$",typeRef);
162        //System.out.println("result: "+ result);
163        int idx = 0;
164        for (Rectangle rect : result) {
165            assertEquals((idx * idx), rect.getSize(), 
166                         "deser reread size failed");
167            idx++;
168        }
169    }

171    @Test
172    public void testDeSerToORGJSONCollectionObject() throws Exception {
173        // test array
174         List<Bean> beanResults = new ArrayList<Bean> ( );
175         Bean tu = new Bean();
176         tu.setName("jim jar");
177         beanResults.add(tu);
178         Bean tu2 = new Bean();
179         tu2.setName("jim2 jar2");
180         tu2.setAge(45);
181         beanResults.add(tu2);
182         
183         String[] filterAttr = {"age","name" };   
184         String filteredSerList = sc.serializeOnlyFilter(beanResults, Bean.class, filterAttr);
185         logger.debug("serList: "+ filteredSerList);
186         
187         sc.addAdapter(null, null,new JsonOrgModule());
188
189         TypeRef<JSONArray> typeRef = new TypeRef<JSONArray>() { };
190         JSONArray jsonOrgResult = JsonPath.parse(filteredSerList).read("$",typeRef);
191         
192         logger.debug("jsonOrgResult: "+ jsonOrgResult.toString(2));
193         assertEquals("jim jar", ((JSONObject)(jsonOrgResult.get(0))).get("name"), 
194                         "DeSer failed name" );
195         assertEquals(45, ((JSONObject)(jsonOrgResult.get(1))).get("age"),
196                      "DeSer failed age" );
197    }
