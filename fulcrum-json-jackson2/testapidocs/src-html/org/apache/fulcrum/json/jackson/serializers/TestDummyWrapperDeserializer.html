022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.List;
025
026import com.fasterxml.jackson.core.JsonParser;
027import com.fasterxml.jackson.core.JsonProcessingException;
028import com.fasterxml.jackson.core.JsonToken;
029import com.fasterxml.jackson.databind.DeserializationContext;
030import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
031
032public class TestDummyWrapperDeserializer<T> extends StdDeserializer<List<T>> {
033
034    /**
035         * 
036         */
037    private static final long serialVersionUID = 1L;
038
039
040    public TestDummyWrapperDeserializer( Class<T> vc )
041    {
042        super( vc );
043    }
044    
045    @Override
046    public List<T> deserialize( JsonParser jp, DeserializationContext ctxt )
047        throws IOException, JsonProcessingException
048    {
049        ArrayList<T> list = new ArrayList<T>();
050        while ((jp.nextToken() != JsonToken.END_OBJECT)) {
051          list.add( (T) jp.getCurrentToken() );
052        }
053      return list;
054    }
055}
