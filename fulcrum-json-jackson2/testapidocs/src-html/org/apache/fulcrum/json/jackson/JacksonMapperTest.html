329    @Test
330    public void testCollectionWithMixins() throws Exception {
331        List<Object> components = new ArrayList<Object>();
332        components.add(new Rectangle(25, 3));
333        components.add(new Rectangle(250, 30));
334        for (int i = 0; i < 3; i++) {
335            Bean filteredBean = new Bean();
336            filteredBean.setName("joe" + i);
337            filteredBean.setAge(i);
338            components.add(filteredBean);
339        }
340
341        sc.addAdapter("M4RMixin", Rectangle.class, RectangleMixin.class).addAdapter(
342                "M4BeanRMixin", Bean.class, BeanMixin.class);
343        String serRect = sc.ser(components);
344        assertEquals(
345                "[{'width':25},{'width':250},{'name':'joe0'},{'name':'joe1'},{'name':'joe2'}]",
346                serRect.replace('"', '\''),  "DeSer failed ");
347        
348        // adding h and name for first two items, adding width for beans
349        String deSerTest = "[{\"width\":25,\"age\":99, \"h\":50,\"name\":\"rect1\"},{\"width\":250,\"name\":\"rect2\"},{\"name\":\"joe0\"},{\"name\":\"joe1\"},{\"name\":\"joe2\"}]";
350        
351        List<Rectangle> typeRectList = new ArrayList<>(); //empty
352        // could not use Mixins here, but Adapters are still set
353        Collection<Rectangle> resultList0 =  sc.deSerCollection(deSerTest, typeRectList, Rectangle.class);
354        logger.debug("resultList0 class:" +resultList0.getClass());
355        for (int i = 0; i < 5; i++) {
356            // name and h should be null as it is ignored,  cft. Mixin
357            assertTrue(((List<Rectangle>)resultList0).get(i).getName()==null);
358            assertTrue(((List<Rectangle>)resultList0).get(i).getH()==0);
359        }
360        // could not use Mixins here, but Adapters are still set
361        Collection<Bean> resultList1 =  sc.deSerCollection(deSerTest, typeRectList, Bean.class);
362        logger.debug("resultList1 class:" +resultList1.getClass());
363        for (int i = 0; i < 5; i++) {
364            logger.debug("resultList1 "+i+ " name:"+((List<Bean>)resultList1).get(i).getName());
365            // name should NOT be null, age should be ignored, cft. BeanMixin
366            assertTrue(((List<Bean>)resultList1).get(i).getName()!=null);
367            assertTrue(((List<Bean>)resultList1).get(i).getAge()==-1);
368        }
369        ((Initializable)sc).initialize();// reinit to default settings
370        Collection<Rectangle> resultList3 =  sc.deSerCollection(deSerTest, typeRectList, Rectangle.class);
371        // h should be set again without Mixin
372        assertTrue(((List<Rectangle>)resultList3).get(0).getH()!=0);
373        for (int i = 0; i < 5; i++) {
374            // name should be set without Mixin
375            assertTrue(((List<Rectangle>)resultList3).get(i).getName()!=null);
376        }
377    }
