138        private String doJob(String... filtereIds) {
139            final int randInt = randInt(999, 1001);
140            List<Bean> tasks = getBeans(randInt);
141            String result = getJson(tasks, Bean.class, true, filtereIds);
142            System.out.println("ser result (id=" + Thread.currentThread() + "):" + result);
143            for (int i = 0; i < filtereIds.length; i++) {
144                assertTrue("Result does not contain (" + filtereIds[i] + ")",
145                        result.contains("\"" + filtereIds[i] + "\""));
146            }
147            return result;
148        }
149
150        private String doFilteredJob(ObjectMapper mapper, String... filtereIds) throws JsonProcessingException {
151            final int randInt = randInt(999, 1001);
152            List<Bean> tasks = getBeans(randInt);
153            String result = serFiltered(tasks, Bean.class, mapper, filtereIds);
154            System.out.println("ser result (id=" + Thread.currentThread() + "):" + result);
155            for (int i = 0; i < filtereIds.length; i++) {
156                assertTrue("Result does not contain (" + filtereIds[i] + ")",
157                        result.contains("\"" + filtereIds[i] + "\""));
158            }
159            return result;
160        }
161
162        private String serFiltered(List<?> list, Class<?> filterClass, ObjectMapper mapper, String[] filterAttr)
163                throws JsonProcessingException {
164            Collection<AnnotationIntrospector> ais = ((AnnotationIntrospectorPair) mapper.getSerializationConfig()
165                    .getAnnotationIntrospector()).allIntrospectors();
166            for (AnnotationIntrospector ai : ais) {
167                if (ai instanceof SimpleNameIntrospector) {
168                    // activate filtering
169                    ((SimpleNameIntrospector) ai).setFilteredClasses(filterClass);
170                }
171            }
172            PropertyFilter pf = SimpleBeanPropertyFilter.filterOutAllExcept(filterAttr);
173            SimpleFilterProvider filter = new SimpleFilterProvider();
174            filter.setDefaultFilter(pf);
175            return mapper.writer(filter).writeValueAsString(list);
176        }
177
178        private ObjectMapper customMapper(boolean withType) {
179            // without copy test fails probably in testsRunInParallel
180            ObjectMapper objectMapper = null;
181            if (withType) {
182                // without copy tests will fail in testsRunInParallel
183                objectMapper = ((Jackson2MapperService) jsonService).getMapper().copy();
184                objectMapper.enableDefaultTypingAsProperty(DefaultTyping.NON_FINAL, "type");
185                AnnotationIntrospector ai = objectMapper.getSerializationConfig().getAnnotationIntrospector();
186                AnnotationIntrospector siai = new SimpleNameIntrospector();
187                AnnotationIntrospector pair = new AnnotationIntrospectorPair(siai, ai);
188                objectMapper.setAnnotationIntrospector(pair);
189            } else {
190                objectMapper = ((Jackson2MapperService) jsonService).getMapper();
191            }
192            return objectMapper;
193        }
194
195        private List<Bean> getBeans(int randNum) {
196            List<Bean> beans = new ArrayList<Bean>();
197            for (int i = 0; i < 50; i++) {
198                Bean dct = new Bean();
199                dct.setName("title" + i);
200                dct.setAge(randNum + i);
201                dct.profession = "prof" + i;
202                beans.add(dct);
203            }
204            return beans;
205        }
206
207        private String getJson(List<?> list, Class clazz, Boolean refresh, String... props) {
208            String result = "";
209            try {
210                result = jsonService.serializeOnlyFilter(list, clazz, refresh, props);
211            } catch (Exception e) {
212                System.out.println(e.getMessage());
213                fail();
214            }
215            return result;
216        }

219    @Test
220    public void testsRunInParallel() {
221        // Parallel among methods in a class in one thread pool
222        Result result = JUnitCore.runClasses(ParallelComputer.methods(), JSONBeansTests.class);
223        System.out.println("fOne:" + fOne);
224        System.out.println("fTwo:" + fTwo);
225        System.out.println("fThree:" + fThree);
226        assertNotNull("Parallel Init exist" + getFailures(result), result.getFailureCount() == 0);
227        assertTrue("Parallel Init" + getTrace(result), result.wasSuccessful());

229        assertNotNull(fOne);
230        assertNotNull(fTwo);
231        assertNotNull(fThree);

241    public static String getFailures(Result result) {
242        List<Failure> failures = result.getFailures();
243        StringBuffer sb = new StringBuffer();
244        for (Failure failure : failures) {
245            sb.append(failure.getMessage());
246            // System.out.println(failure.getMessage());
247        }
248        return sb.toString();
249    }
250
251    public static String getTrace(Result result) {
252        List<Failure> failures = result.getFailures();
253        StringBuffer sb = new StringBuffer();
254        for (Failure failure : failures) {
255            failure.getException().printStackTrace();
256            if (failure.getException().getCause() != null)
257                failure.getException().getCause().printStackTrace();
258            sb.append(failure.getTrace());
259        }
260        return sb.toString();
261    }
