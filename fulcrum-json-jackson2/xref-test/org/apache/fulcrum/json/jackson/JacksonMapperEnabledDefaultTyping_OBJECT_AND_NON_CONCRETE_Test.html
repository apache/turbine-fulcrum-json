103     public void testSerializeDeSerializeDate() throws Exception {
104         Map<String, Date> map = new HashMap<String, Date>();
105         Calendar sourceDate = Calendar.getInstance();
106         sourceDate.set(1999, 3, 10);
107         
108 //        logger.debug("sourceDate calendar:"+ sourceDate);
109         logger.debug("sourceDate date:"+ sourceDate.getTime());
110         logger.debug("sourceDate millisec:"+ sourceDate.getTime().getTime());
111         map.put("mydate",sourceDate.getTime());
112         map.put("mydate2",Calendar.getInstance().getTime());
113         // default dateformat dd/mm/yy -> day time will be cut off !(hh, mm)
114         // first serialize
115         String serJson0 =  sc.ser(map, false);
116         String serJson =  sc.ser(map, Map.class, false);
117         
118         logger.debug("serJson:"+ serJson0);
119         assertEquals(serJson0, serJson);
120          //second deserialize 
121         DateKeyMixin serObject =sc.deSer(serJson0, DateKeyMixin.class);
122         assertTrue(serObject.mydate instanceof Date);
123         
124         logger.debug("resultDate millisec: " + ((Date)serObject.mydate).getTime() +" source:"+ sourceDate.getTime().getTime() );
125         // cleanup all values the mapper dateformat, which is MM/dd/yyyy, does not contain.
126         sourceDate.set(Calendar.HOUR, 0);
127         sourceDate.set(Calendar.MINUTE, 0);
128         sourceDate.set(Calendar.SECOND, 0);
129         sourceDate.set(Calendar.MILLISECOND, 0);
130         sourceDate.set(Calendar.HOUR_OF_DAY, 0);
131         assertEquals(
132                 ((Date)serObject.mydate).getTime(),sourceDate.getTime().getTime(),
133                 "millisec of result and source date should be equal, after zeroing not used formatter values:: ");
134     }

152         
153         String serJson0 =  sc.ser(map, false);
154         String serJson =  sc.ser(map, Map.class, false);

157         assertEquals(serJson0, serJson);

162                 ((Date)serObject.mydate).getTime() +" source:"+ sourceDate.getTime().getTime() );
163 
164         logger.debug("may not be equal: millisec(resultDate(string)):" + ((Date)serObject.mydate).getTime() + 
165                 " millisec(sourceDate):"+ sourceDate.getTime().getTime() );
166         // cleanup all values the mapper dateformat does not contain.
167         sourceDate.set(Calendar.SECOND, 0);
168         sourceDate.set(Calendar.MILLISECOND, 0);
169         assertEquals(
