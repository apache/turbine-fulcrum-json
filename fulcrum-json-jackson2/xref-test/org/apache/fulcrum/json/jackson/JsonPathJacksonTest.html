64      @BeforeEach
65      public void setUp() throws Exception {
66          sc = (JsonService) this.lookup(JsonService.ROLE);
67          logger = new Log4JLogger(LogManager.getLogger(getClass().getName()) );
68          try {
69              Configuration conf = Configuration.defaultConfiguration();
70              logger.debug("jayway jsonpath conf:"+ conf.jsonProvider());
71              assertEquals( JacksonJsonProvider.class.getName(), conf.jsonProvider().getClass().getName(),
72                            "Jackson JsonPath JsonProvider match failed ");
73              
74              logger.debug("Jackson2MapperService.mapper:"+ ((Jackson2MapperService)sc).getMapper()  + " confjsonProvider:" + conf.jsonProvider());
75              
76              assertTrue(conf.jsonProvider() instanceof JacksonJsonProvider, 
77                         "JsonProvider is not a JacksonJsonProvider ");
78              
79              assertEquals(((Jackson2MapperService)sc).getMapper(), ((JacksonJsonProvider)conf.jsonProvider()).getObjectMapper(), 
80                           "JacksonJsonProvider mapper is not configured mapper");
81          } catch (Throwable e) {
82              if (e.getCause() != null && e.getCause() instanceof ClassNotFoundException) { 
83                  logger.error(e.getMessage(), e.getCause());
84                  fail("Check correct initialization with useJsonPath = true):");
85              } else {
86                  fail(e.getMessage());
87              }
88          }

100     @Test
101     public void testSerializeDateWithDefaultDateFormat() throws Exception {
102         Map<String, Date> map = new HashMap<String, Date>();
103         Calendar refDate = Calendar.getInstance();
104         map.put("date", refDate.getTime());
105         String serJson = sc.ser(map);
106         //System.out.println("serJson date: "+ serJson);
107         Date date = JsonPath.parse(serJson).read("$.date", Date.class);// .using(conf)
108         Calendar parsedDate = Calendar.getInstance();
109         parsedDate.setTime(date);
110         assertEquals(refDate.get(Calendar.DATE), parsedDate.get(Calendar.DATE), "Serialization failed ");
111     }
