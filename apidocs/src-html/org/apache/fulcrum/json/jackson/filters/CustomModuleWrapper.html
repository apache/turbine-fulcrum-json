022import org.apache.fulcrum.json.jackson.CustomModule;
023
024import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
025import com.fasterxml.jackson.databind.ser.std.StdSerializer;
026
027/**
028 * Provides a wrapper module as a helper class for the inner class {@link CustomModule}.
029 * 
030 * @author gkallidis
031 *
032 * @param <T> the class to wrap
033 */
034public class CustomModuleWrapper<T> {
035        
036        // serializer and deserializer objects
037    StdSerializer<T> ser = null;
038    StdDeserializer<T> deSer = null;
039
040    /**
041     * Constructor for the custom module wrapper
042     * 
043     * @param ser the standard serializer
044     * @param deSer the standard de-serializer
045     */
046    public CustomModuleWrapper(StdSerializer<T> ser, StdDeserializer<T> deSer) {
047        this.ser = ser;
048        this.deSer = deSer;
049    }
050
051    /**
052     * @return the serializer
053     */
054    public StdSerializer<T> getSer() {
055        return ser;
056    }
057
058    /**
059     * @param ser set the standard serializer
060     */
061    public void setSer(StdSerializer<T> ser) {
062        this.ser = ser;
063    }
064
065    /**
066     * @return the de-serializer in use
067     */
068    public StdDeserializer<T> getDeSer() {
069        return deSer;
070    }
071
072    /**
073     * @param deSer standard deserializer
074     */
075    public void setDeSer(StdDeserializer<T> deSer) {
076        this.deSer = deSer;
077    }
078
079}
